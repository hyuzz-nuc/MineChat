## 一、部署架构

```
                    ┌──────────────┐
                    │   Nginx      │
                    │   (反向代理)  │
                    │   :443/80    │
                    └──────┬───────┘
                           │
              ┌────────────┼────────────┐
              │                         │
     ┌────────▼────────┐     ┌─────────▼───────┐
     │  前端静态文件    │     │  后端 Node.js    │
     │  (Nginx serve)  │     │  (PM2管理)       │
     │  :80/443        │     │  :3000           │
     └─────────────────┘     └─────────┬───────┘
                                       │
                          ┌────────────┼────────────┐
                          │                         │
                 ┌────────▼────────┐     ┌─────────▼───────┐
                 │  PostgreSQL 15  │     │  Redis 7        │
                 │  :5432          │     │  :6379          │
                 └─────────────────┘     └─────────────────┘
```

## 三、部署步骤

### 3.1 系统准备

```bash
# 更新系统
sudo apt update && sudo apt upgrade -y

# 安装基础工具
sudo apt install -y curl git build-essential

# 安装 Node.js 18+
curl -fsSL https://deb.nodesource.com/setup_18.x | sudo -E bash -
sudo apt install -y nodejs

# 安装 pnpm
npm install -g pnpm

# 安装 PM2（进程管理）
npm install -g pm2

# 安装 Nginx
sudo apt install -y nginx

# 安装 PostgreSQL 15
sudo sh -c 'echo "deb http://apt.postgresql.org/pub/repos/apt $(lsb_release -cs)-pgdg main" > /etc/apt/sources.list.d/pgdg.list'
curl -fsSL https://www.postgresql.org/media/keys/ACCC4CF8.asc | sudo gpg --dearmor -o /etc/apt/trusted.gpg.d/postgresql.gpg
sudo apt update
sudo apt install -y postgresql-15

# 安装 Redis
sudo apt install -y redis-server
```

### 3.2 数据库配置

```bash
# 创建数据库和用户
sudo -u postgres psql
CREATE USER minechat WITH PASSWORD '你的密码';
CREATE DATABASE minechat OWNER minechat;
GRANT ALL PRIVILEGES ON DATABASE minechat TO minechat;
\q
```

### 3.3 项目部署

```bash
# 克隆项目
cd /opt
git clone git@github.com:hyuzz-nuc/MineChat.git
cd MineChat

# ── 后端 ──
cd server
pnpm install
pnpm approve-builds  # 授权原生模块

# 配置环境变量
cp .env.example .env
# 编辑 .env，填入生产环境配置：
# - DATABASE_URL=postgresql://minechat:密码@localhost:5432/minechat
# - JWT_SECRET=强随机密钥
# - JWT_REFRESH_SECRET=强随机刷新密钥
# - CLIENT_URL=https://你的域名
# - NODE_ENV=production

# 初始化数据库
pnpm db:generate
pnpm db:push
pnpm db:seed  # 可选：填充测试数据

# 编译TypeScript
pnpm build

# ── 前端 ──
cd ../client
pnpm install

# 配置环境变量
# 编辑 .env：
# VITE_API_BASE_URL=https://你的域名/api/v1
# VITE_SOCKET_URL=https://你的域名

# 构建生产版本
pnpm build
```

### 3.4 PM2 配置

```bash
# 创建 PM2 生态文件
cd /opt/MineChat/server
```

创建 `ecosystem.config.js`：

```javascript
module.exports = {
  apps: [{
    name: 'minechat-server',
    script: 'dist/app.js',
    cwd: '/opt/MineChat/server',
    instances: 1,
    autorestart: true,
    watch: false,
    max_memory_restart: '500M',
    env: {
      NODE_ENV: 'production',
      PORT: 3000,
    },
  }],
};
```

```bash
# 启动
pm2 start ecosystem.config.js

# 设置开机自启
pm2 startup
pm2 save
```

### 3.5 Nginx 配置

创建 `/etc/nginx/sites-available/minechat`：

```nginx
server {
    listen 80;
    server_name 你的域名;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name 你的域名;

    # SSL 证书（用 Let's Encrypt）
    ssl_certificate /etc/letsencrypt/live/你的域名/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/你的域名/privkey.pem;

    # 前端静态文件
    root /opt/MineChat/client/dist;
    index index.html;

    # 前端路由（SPA）
    location / {
        try_files $uri $uri/ /index.html;
    }

    # API 反向代理
    location /api/ {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # WebSocket 反向代理
    location /socket.io/ {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 86400;
    }

    # 静态资源缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff2?)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
}
```

```bash
# 启用站点
sudo ln -s /etc/nginx/sites-available/minechat /etc/nginx/sites-enabled/
sudo nginx -t
sudo systemctl reload nginx

# SSL 证书
sudo apt install certbot python3-certbot-nginx
sudo certbot --nginx -d 你的域名
```

## 四、更新部署

```bash
cd /opt/MineChat

# 拉取最新代码
git pull origin main

# 更新后端
cd server
pnpm install
pnpm build
pm2 restart minechat-server

# 更新前端
cd ../client
pnpm install
pnpm build
# Nginx 自动serve新的dist文件
```
